save
	:w		:file save
	:w filename	:save file in filename(a duplicated form)
	:w! filename	:save file in filename	
	:wq		:file save and quit
	:wq! filename	:save file in filename(a duplicated form) and quit

	:q!	:file quit(force to quit,without save)
